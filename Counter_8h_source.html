73  void MakeHistogram(TH1* h) const
74  {
75  LOG(INFO) << "Making Histogram " << h->GetName() << " to accomodate counter of size " << Tc::size;
76  TAxis* axis = h->GetXaxis();
77  if (((UInt_t)axis->GetNbins()) < Tc::size) {
78  LOG(FATAL) << "The histogram size (" << axis->GetNbins() << ") is not large enough to accomodate the counter size (" << Tc::size << ")";
79  }
80  h->Reset();
81  UInt_t histo_size = Tc::size;
82  for (UInt_t i = 0; i < Tc::size; i++) {
83  if (Tc::names[i].IsNull()) {
84  histo_size--;
85  }
86  }
87  if (histo_size == 0) {
88  LOG(FATAL) << "Asked to produce a histogram with size " << histo_size << ", check counter bin labels";
89  }
90  axis->Set(histo_size, 0, histo_size);
91  UInt_t binx = 1;
92  for (UInt_t i = 0; i < Tc::size; i++) {
93  if (Tc::names[i].IsNull()) {
94  continue;
95  }
96  LOG(INFO) << "Setting bin " << binx << "/" << histo_size << " to contain counter for " << Tc::names[i] << " (index " << i << "/" << Tc::size - 1 << ")";
97  if (histo_size < binx) {
98  LOG(FATAL) << "Making bin outside of histogram limits!";
99  }
100  axis->SetBinLabel(binx++, Tc::names[i]);
101  }
102  h->Reset();
103 #ifdef ENABLE_PRINT_HISTOGRAMS_MODE
104  h->Print("All");
105 #endif
106  }

112  void FillHistogram(TH1* h, UInt_t biny = 0, UInt_t binz = 0) const
113  {
114  LOG(INFO) << "Filling Histogram " << h->GetName() << " with counter contents";
115  UInt_t binx = 1;
116  const UInt_t nbinsx = h->GetNbinsX();
117  for (UInt_t i = 0; i < Tc::size; i++) {
118  if (Tc::names[i].IsNull()) {
119  if (counter[i] > 0) {
120  LOG(WARNING) << "Counter at position " << i << " was non empty (" << counter[i] << ") but was discarded because of empty lables";
121  }
122  continue;
123  }
124 #ifdef ENABLE_COUNTER_DEBUG_MODE
125  LOG(INFO) << "Filling bin " << binx << " of position " << i << " of label " << Tc::names[i] << " with " << counter[i];
126 #endif
127  if (binx > nbinsx) {
128  LOG(FATAL) << "Filling histogram " << h->GetName() << " at position " << binx << " i.e. past its size (" << nbinsx << ")!";
129  }
130  if (!Tc::names[i].EqualTo(h->GetXaxis()->GetBinLabel(binx))) {
131  LOG(FATAL) << "Bin" << binx << " does not have the expected label '" << h->GetXaxis()->GetBinLabel(binx) << "' vs '" << Tc::names[i] << "'";
132  }
133  if (biny > 0) {
134  if (binz > 0) {
135  h->SetBinContent(binx, biny, binz, counter[i]);
136  } else {
137  h->SetBinContent(binx, biny, counter[i]);
138  }
139  } else {
140  h->SetBinContent(binx, counter[i]);
141  }
142  binx++;
143  }
144 #ifdef ENABLE_PRINT_HISTOGRAMS_MODE
145  h->Print("All");
146 #endif
147  }
